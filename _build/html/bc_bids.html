def bid_nt(hd):
   bid = False
   if (notrump(hd) ):
      bid = 0
      pt = hcp(hd)
      match pt:
         case _ if pt <= 17:
            bid = '1NT'
         case _ if (17 < pt <= 20):
            bid = '2NT'
         case _ if pt >= 21:
            bid = '3NT'
   return bid

def bid_std(hd):
   bid = False
   if (standard(hd) and twosuiter(hd) ):
      bid = 0
   temp = make_bid_table(hd)
   s = temp.sort(4, True).column(0).item(0)
   match s:
      case 'spades' :
         bid = '1S'
      case 'hearts' :
         bid = '1H'
      case 'diamonds' :
         bid = '1D'
      case 'clubs' :
         bid = False
   return bid
